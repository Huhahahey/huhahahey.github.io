TypeScript 高级类型编程技巧.

TypeScript 类型系统的力量

TypeScript 的类型系统不仅仅是静态检查,它是一门图灵完备的类型编程语言。掌握高级类型技巧,可以让我们在编译期捕获更多错误,提升代码质量。

泛型编程进阶

基础泛型约束

// 基础泛型
function identity<T>(arg: T): T {
  return arg
}

// 带约束的泛型
interface Lengthwise {
  length: number
}

function loggingIdentity<T extends Lengthwise>(arg: T): T {
  console.log(arg.length) // 现在我们知道它有 .length 属性
  return arg
}

// 使用示例
loggingIdentity("hello") // ✅ 字符串有 length
loggingIdentity([1, 2, 3]) // ✅ 数组有 length
loggingIdentity(123) // ❌ 数字没有 length

多泛型参数

// 交换元组的两个元素类型
function swap<T, U>(tuple: [T, U]): [U, T] {
  return [tuple[1], tuple[0]]
}

const result = swap(["hello", 42]) // 类型: [number, string]

// 合并两个对象
function mergeObjects<T extends object, U extends object>(
  obj1: T,
  obj2: U
): T & U {
  return { ...obj1, ...obj2 }
}

const merged = mergeObjects(
  { name: "Alice" },
  { age: 25 }
) // 类型: { name: string } & { age: number }

泛型默认值

interface Response<T = any> {
  data: T
  status: number
  message: string
}

// 使用默认值
const response1: Response = {
  data: null,
  status: 200,
  message: "OK"
}

// 指定具体类型
const response2: Response<User> = {
  data: { id: 1, name: "Alice" },
  status: 200,
  message: "OK"
}

条件类型

条件类型让我们可以根据类型关系进行选择:

// 基础条件类型
type IsString<T> = T extends string ? true : false

type A = IsString<string> // true
type B = IsString<number> // false

// 提取函数返回类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never

function getUser() {
  return { id: 1, name: "Alice" }
}

type User = ReturnType<typeof getUser> // { id: number; name: string }

// 提取 Promise 的值类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T

type UserId = UnwrapPromise<Promise<number>> // number
type PlainId = UnwrapPromise<number> // number

嵌套条件类型

// 扁平化嵌套数组
type Flatten<T> = T extends Array<infer U>
  ? U extends Array<any>
    ? Flatten<U>
    : U
  : T

type NestedArray = number[][][]
type Flat = Flatten<NestedArray> // number

映射类型

映射类型可以基于已有类型创建新类型:

基础映射

// 将所有属性变为可选
type Partial<T> = {
  [P in keyof T]?: T[P]
}

// 将所有属性变为只读
type Readonly<T> = {
  readonly [P in keyof T]: T[P]
}

// 使用示例
interface User {
  id: number
  name: string
  email: string
}

type PartialUser = Partial<User>
// {
//   id?: number;
//   name?: string;
//   email?: string;
// }

type ReadonlyUser = Readonly<User>
// {
//   readonly id: number;
//   readonly name: string;
//   readonly email: string;
// }

键的重映射

// 将属性名转为大写
type UppercaseKeys<T extends object> = {
  [K in keyof T as Uppercase<string & K>]: T[K]
}

type Config = {
  host: string
  port: number
}

type UpperConfig = UppercaseKeys<Config>
// {
//   HOST: string;
//   PORT: number;
// }

// 过滤特定类型的属性
type StringKeys<T> = {
  [K in keyof T as T[K] extends string ? K : never]: T[K]
}

type OnlyStrings = StringKeys<{
  name: string
  age: number
  email: string
}>
// {
//   name: string;
//   email: string;
// }

深度映射

// 深度将所有属性变为可选
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P]
}

interface Address {
  street: string
  city: string
}

interface UserProfile {
  user: {
    name: string
    age: number
  }
  address: Address
}

type DeepPartialProfile = DeepPartial<UserProfile>
// {
//   user?: {
//     name?: string;
//     age?: number;
//   };
//   address?: {
//     street?: string;
//     city?: string;
//   };
// }

工具类型实战

Pick 和 Omit

// Pick: 选择部分属性
type Pick<T, K extends keyof T> = {
  [P in K]: T[P]
}

// Omit: 排除部分属性
type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>

// 使用示例
interface User {
  id: number
  name: string
  email: string
  password: string
}

// 只选择公开信息
type PublicUser = Pick<User, 'id' | 'name'>
// {
//   id: number;
//   name: string;
// }

// 排除敏感信息
type SafeUser = Omit<User, 'password'>
// {
//   id: number;
//   name: string;
//   email: string;
// }

Record

// 创建键值对映射类型
type Record<K extends keyof any, T> = {
  [P in K]: T
}

// 使用示例
type Role = 'admin' | 'user' | 'guest'

interface Permissions {
  canRead: boolean
  canWrite: boolean
  canDelete: boolean
}

const rolePermissions: Record<Role, Permissions> = {
  admin: { canRead: true, canWrite: true, canDelete: true },
  user: { canRead: true, canWrite: true, canDelete: false },
  guest: { canRead: true, canWrite: false, canDelete: false }
}

Extract 和 Exclude

// Extract: 提取可分配给 U 的类型
type Extract<T, U> = T extends U ? T : never

// Exclude: 排除可分配给 U 的类型
type Exclude<T, U> = T extends U ? never : T

// 使用示例
type EventTypes = 'click' | 'scroll' | 'mousemove' | 'keydown'

type MouseEvents = Extract<EventTypes, `mouse${string}`>
// 'mousemove'

type KeyboardEvents = Exclude<EventTypes, MouseEvents>
// 'click' | 'scroll' | 'keydown'

模板字面量类型

TypeScript 4.1 引入的强大特性:

// 基础用法
type Greeting = `Hello ${string}`

const greet: Greeting = "Hello World" // ✅
const greet2: Greeting = "Hi World" // ❌

// 事件名称生成
type EventName = 'click' | 'hover' | 'focus'
type Element = 'button' | 'input' | 'div'

type EventHandler = `on${Capitalize<EventName>}${Capitalize<Element>}`
// 'onClickButton' | 'onClickInput' | ... | 'onFocusDiv'

// CSS 属性组合
type Direction = 'top' | 'right' | 'bottom' | 'left'
type Size = 'sm' | 'md' | 'lg'

type MarginClass = `m-${Direction}-${Size}`
// 'm-top-sm' | 'm-top-md' | ... | 'm-left-lg'

字符串操作工具

// 内置字符串工具类型
type Lowercase<T extends string> = ...
type Uppercase<T extends string> = ...
type Capitalize<T extends string> = ...
type Uncapitalize<T extends string> = ...

// 使用示例
type Result1 = Lowercase<'HELLO'> // 'hello'
type Result2 = Uppercase<'hello'> // 'HELLO'
type Result3 = Capitalize<'hello'> // 'Hello'
type Result4 = Uncapitalize<'Hello'> // 'hello'

高级实战案例

类型安全的 API 客户端

// 定义 API 端点
interface ApiEndpoints {
  '/users': {
    GET: { response: User[]; query: { page?: number } }
    POST: { response: User; body: CreateUserInput }
  }
  '/users/:id': {
    GET: { response: User; params: { id: number } }
    PUT: { response: User; params: { id: number }; body: UpdateUserInput }
    DELETE: { response: void; params: { id: number } }
  }
}

// 提取路径参数
type PathParams<Path extends string> = 
  Path extends `${string}:${infer Param}/${infer Rest}`
    ? { [K in Param | keyof PathParams<Rest>]: string | number }
    : Path extends `${string}:${infer Param}`
    ? { [K in Param]: string | number }
    : {}

// API 客户端类型
type ApiClient<Endpoints> = {
  [Path in keyof Endpoints]: {
    [Method in keyof Endpoints[Path]]: (
      args: {
        params?: PathParams<Path & string>
        query?: Endpoints[Path][Method] extends { query: infer Q } ? Q : never
        body?: Endpoints[Path][Method] extends { body: infer B } ? B : never
      }
    ) => Promise<Endpoints[Path][Method] extends { response: infer R } ? R : never>
  }
}

// 使用示例
const api: ApiClient<ApiEndpoints> = createApiClient('/api')

// 类型安全,自动补全
const users = await api['/users'].GET({ query: { page: 1 } })
const user = await api['/users/:id'].GET({ params: { id: 1 } })

表单验证类型

// 验证规则
type ValidationRule = 
  | { type: 'required'; message: string }
  | { type: 'minLength'; value: number; message: string }
  | { type: 'maxLength'; value: number; message: string }
  | { type: 'pattern'; value: RegExp; message: string }
  | { type: 'email'; message: string }

// 表单字段配置
type FieldConfig = {
  rules: ValidationRule[]
  defaultValue?: any
}

// 表单 schema
type FormSchema = Record<string, FieldConfig>

// 从 schema 推断表单值类型
type FormValues<Schema extends FormSchema> = {
  [K in keyof Schema]: Schema[K]['defaultValue'] extends undefined
    ? string
    : Schema[K]['defaultValue']
}

// 使用示例
const loginFormSchema = {
  email: {
    rules: [
      { type: 'required', message: '邮箱不能为空' },
      { type: 'email', message: '请输入有效的邮箱地址' }
    ]
  },
  password: {
    rules: [
      { type: 'required', message: '密码不能为空' },
      { type: 'minLength', value: 6, message: '密码至少6位' }
    ],
    defaultValue: ''
  }
} as const

type LoginFormValues = FormValues<typeof loginFormSchema>
// {
//   email: string;
//   password: string;
// }

性能优化建议

1. 避免过度复杂的类型

// ❌ 避免:过于复杂的嵌套条件类型
type ComplexType<T> = T extends A 
  ? T extends B 
    ? T extends C 
      ? ... 
      : ...
    : ...
  : ...

// ✅ 推荐:拆分为多个简单的类型
type Step1<T> = T extends A ? TrueType : FalseType
type Step2<T> = T extends B ? TrueType : FalseType
type FinalType<T> = Step1<T> extends Step2<T> ? ... : ...

2. 使用类型缓存

// 缓存常用类型转换
type CachedToString<T> = T extends string ? T : string

// 避免重复计算
type Result1 = CachedToString<number>
type Result2 = CachedToString<string>

3. 合理使用 any 和 unknown

// ❌ 避免:滥用 any
function process(data: any): any {
  return data
}

// ✅ 推荐:使用 unknown 并进行类型守卫
function process(data: unknown): string {
  if (typeof data === 'string') {
    return data.toUpperCase()
  }
  throw new Error('Invalid input')
}

总结

TypeScript 的高级类型系统为我们提供了强大的工具:

  1. 泛型: 实现类型安全的代码复用
  2. 条件类型: 根据类型关系进行分支
  3. 映射类型: 基于现有类型创建新类型
  4. 模板字面量类型: 类型级别的字符串操作
  5. 工具类型: 常用的类型转换助手

掌握这些技巧,你可以:

  • 在编译期捕获更多错误
  • 提供更好的 IDE 智能提示
  • 编写更灵活、更可复用的代码
  • 构建类型安全的库和框架

记住,类型系统是为你服务的工具,不要为了炫技而过度复杂化。保持类型定义的清晰和可维护性同样重要!
